#Build with RouterOS 6.48
#Testing platform is a CHR within Virtualbox

/interface bridge add name=BR_LAN
/interface bridge add name=BR_MNGT
/interface bridge add name=BR_WAN
/interface list add name=WAN
/interface list add name=LAN
/interface list add name=MNGT
/interface wireless security-profiles set [ find default=yes ] supplicant-identity=MikroTik
/interface bridge port add bridge=BR_MNGT interface=ether4 multicast-router=disabled
/interface bridge port add bridge=BR_WAN interface=ether1 multicast-router=disabled
/interface bridge port add bridge=BR_LAN interface=ether2 multicast-router=disabled
/interface bridge port add bridge=BR_LAN interface=ether3 multicast-router=disabled
/interface bridge settings set use-ip-firewall=yes
/interface list member add interface=BR_WAN list=WAN
/interface list member add interface=BR_LAN list=LAN
/interface list member add interface=BR_MNGT list=MNGT
/ip address add address=192.168.254.254/24 interface=BR_MNGT network=192.168.254.0
/ip dhcp-client add disabled=no interface=BR_WAN
/ip firewall nat add action=masquerade chain=srcnat out-interface=BR_WAN
/ip firewall raw add action=jump chain=prerouting comment="Jump Winbox verkeer naar Winbox Chain" dst-port=8291 jump-target=WINBOX protocol=tcp
/ip firewall raw add action=accept chain=WINBOX comment="Accepteer Winbox verkeer op BR_MNGT" in-interface=BR_MNGT
/ip firewall raw add action=drop chain=WINBOX comment="Drop Winbox als de interface niet BR_MNGT is." in-interface=!BR_MNGT
/ip service set telnet disabled=yes
/ip service set ftp disabled=yes
/ip service set www disabled=yes
/ip service set ssh disabled=yes
/ip service set api disabled=yes
/ip service set api-ssl disabled=yes
/system identity set name=RTR_1
/tool mac-server set allowed-interface-list=MNGT
/tool mac-server mac-winbox set allowed-interface-list=MNGT
/tool mac-server ping set enabled=no
